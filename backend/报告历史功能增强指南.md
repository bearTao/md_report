## 需求说明

### 1. 显示所有状态的报告
当前问题：报告历史模块只显示成功的报告，无法查看失败或进行中的报告。

需求：显示所有状态的报告，包括：
- ✅ **SUCCESS** - 成功
- ⏳ **RUNNING** - 进行中
- 🕐 **PENDING** - 排队中
- ❌ **FAILED** - 失败

### 2. 添加"查看生成过程"按钮
当前问题：无法快速跳转到报告生成进度页面。

需求：添加一个按钮，点击后跳转到对应的生成进度页面：
```
http://localhost:5174/generate/{task_id}
```

## 前端实现指南

### 1. 数据类型定义

```typescript
// types/report.ts

export enum ReportStatus {
  SUCCESS = 'success',
  FAILED = 'failed',
  RUNNING = 'running',
  PENDING = 'pending'
}

export interface ReportListItem {
  id: string;
  template_id: string;
  task_id?: string | null;  // ✨ 新增
  title?: string | null;
  status: ReportStatus;
  created_at: string;
}

export interface ReportListResponse {
  items: ReportListItem[];
  total: number;
}
```

### 2. API 调用

```typescript
// api/reports.ts

import axios from 'axios';
import { ReportListResponse, ReportStatus } from '../types/report';

/**
 * 获取报告列表
 * @param page 页码（从1开始）
 * @param pageSize 每页数量
 * @param status 状态过滤（可选）
 * @param templateId 模板ID过滤（可选）
 */
export async function getReports(
  page: number = 1,
  pageSize: number = 20,
  status?: ReportStatus,
  templateId?: string
): Promise<ReportListResponse> {
  const params: any = { page, page_size: pageSize };
  
  if (status) {
    params.status = status;
  }
  
  if (templateId) {
    params.template_id = templateId;
  }
  
  const response = await axios.get('/api/reports', { params });
  return response.data;
}
```

### 3. 状态图标和颜色

```typescript
// utils/reportStatus.ts

export function getStatusIcon(status: ReportStatus): string {
  switch (status) {
    case ReportStatus.SUCCESS:
      return '✅';
    case ReportStatus.FAILED:
      return '❌';
    case ReportStatus.RUNNING:
      return '⏳';
    case ReportStatus.PENDING:
      return '🕐';
    default:
      return '❓';
  }
}

export function getStatusText(status: ReportStatus): string {
  switch (status) {
    case ReportStatus.SUCCESS:
      return '生成成功';
    case ReportStatus.FAILED:
      return '生成失败';
    case ReportStatus.RUNNING:
      return '生成中';
    case ReportStatus.PENDING:
      return '排队中';
    default:
      return '未知';
  }
}

export function getStatusColor(status: ReportStatus): string {
  switch (status) {
    case ReportStatus.SUCCESS:
      return 'green';
    case ReportStatus.FAILED:
      return 'red';
    case ReportStatus.RUNNING:
      return 'blue';
    case ReportStatus.PENDING:
      return 'orange';
    default:
      return 'gray';
  }
}
```

### 4. 报告历史组件（React示例）

```tsx
// components/ReportHistory.tsx

import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { getReports } from '../api/reports';
import { ReportListItem, ReportStatus } from '../types/report';
import { getStatusIcon, getStatusText, getStatusColor } from '../utils/reportStatus';

export default function ReportHistory() {
  const navigate = useNavigate();
  const [reports, setReports] = useState<ReportListItem[]>([]);
  const [total, setTotal] = useState(0);
  const [page, setPage] = useState(1);
  const [pageSize] = useState(20);
  const [loading, setLoading] = useState(false);
  const [statusFilter, setStatusFilter] = useState<ReportStatus | undefined>();

  // 加载报告列表
  const loadReports = async () => {
    setLoading(true);
    try {
      const data = await getReports(page, pageSize, statusFilter);
      setReports(data.items);
      setTotal(data.total);
    } catch (error) {
      console.error('加载报告列表失败:', error);
    } finally {
      setLoading(false);
    }
  };

  // 页面加载和筛选变化时重新加载
  useEffect(() => {
    loadReports();
  }, [page, statusFilter]);

  // 跳转到生成进度页面
  const handleViewGeneration = (taskId: string) => {
    navigate(`/generate/${taskId}`);
  };

  // 查看报告详情
  const handleViewReport = (reportId: string) => {
    navigate(`/reports/${reportId}`);
  };

  return (
    <div className="report-history">
      <h1>报告历史</h1>

      {/* 状态筛选器 */}
      <div className="filters">
        <label>状态筛选：</label>
        <select 
          value={statusFilter || ''} 
          onChange={(e) => setStatusFilter(e.target.value as ReportStatus || undefined)}
        >
          <option value="">全部</option>
          <option value={ReportStatus.SUCCESS}>成功</option>
          <option value={ReportStatus.FAILED}>失败</option>
          <option value={ReportStatus.RUNNING}>进行中</option>
          <option value={ReportStatus.PENDING}>排队中</option>
        </select>
      </div>

      {/* 报告列表 */}
      {loading ? (
        <div>加载中...</div>
      ) : (
        <table className="report-table">
          <thead>
            <tr>
              <th>状态</th>
              <th>标题</th>
              <th>创建时间</th>
              <th>操作</th>
            </tr>
          </thead>
          <tbody>
            {reports.map((report) => (
              <tr key={report.id}>
                <td>
                  <span style={{ color: getStatusColor(report.status) }}>
                    {getStatusIcon(report.status)} {getStatusText(report.status)}
                  </span>
                </td>
                <td>{report.title || '无标题'}</td>
                <td>{new Date(report.created_at).toLocaleString('zh-CN')}</td>
                <td>
                  {/* 根据状态显示不同的按钮 */}
                  {report.status === ReportStatus.SUCCESS && (
                    <button onClick={() => handleViewReport(report.id)}>
                      查看报告
                    </button>
                  )}
                  
                  {/* ✨ 新增：查看生成过程按钮（所有状态都显示，只要有task_id） */}
                  {report.task_id && (
                    <button 
                      onClick={() => handleViewGeneration(report.task_id!)}
                      className="btn-secondary"
                      style={{ marginLeft: '8px' }}
                    >
                      查看生成过程
                    </button>
                  )}
                  
                  {/* 失败状态可以重试 */}
                  {report.status === ReportStatus.FAILED && (
                    <button 
                      onClick={() => handleViewGeneration(report.task_id!)}
                      className="btn-warning"
                      style={{ marginLeft: '8px' }}
                    >
                      重新生成
                    </button>
                  )}
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      )}

      {/* 分页 */}
      <div className="pagination">
        <button 
          disabled={page === 1}
          onClick={() => setPage(page - 1)}
        >
          上一页
        </button>
        <span>第 {page} 页，共 {Math.ceil(total / pageSize)} 页</span>
        <button 
          disabled={page * pageSize >= total}
          onClick={() => setPage(page + 1)}
        >
          下一页
        </button>
      </div>
    </div>
  );
}
```

<script setup lang="ts">
import { ref, computed, watch } from 'vue';

const router = useRouter();
const reports = ref<ReportListItem[]>([]);
const total = ref(0);
const page = ref(1);
const pageSize = 20;
const loading = ref(false);
const statusFilter = ref<ReportStatus | undefined>();

const totalPages = computed(() => Math.ceil(total.value / pageSize));

  loading.value = true;
  try {
    const data = await getReports(page.value, pageSize, statusFilter.value);

### 1. 自动刷新进行中的报告

```typescript
// 自动刷新进行中和排队中的报告
useEffect(() => {
  // 检查是否有进行中或排队中的报告
  const hasActiveReports = reports.some(r => 
    r.status === ReportStatus.RUNNING || 
    r.status === ReportStatus.PENDING
  );

  if (hasActiveReports) {
    // 每5秒刷新一次
    const interval = setInterval(() => {
      loadReports();
    }, 5000);

    return () => clearInterval(interval);
  }
}, [reports]);
```
